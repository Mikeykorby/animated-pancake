-- Add the memory of the current conversation for a specific player
local function addToMemory(playerName, message, response)
    -- Initialize memory for player if it doesn't exist
    if not memory[playerName] then
        memory[playerName] = {}
    end
    
    local playerMemory = memory[playerName]
    
    if #playerMemory >= MAX_MEMORY then
        table.remove(playerMemory, 1) -- Remove the oldest entry if at limit
    end
    table.insert(playerMemory, {message = message, response = response})
    warn("Memory updated for " .. playerName .. ". Current memory size: " .. #playerMemory)
end

-- Get memory for a specific player
local function getPlayerMemory(playerName)
    return memory[playerName] or {}
end

-- Reset memory function (can reset all or specific player)
local function resetMemory(playerName)
    if playerName then
        memory[playerName] = {}
        warn("Memory has been reset for " .. playerName .. "!")
    else
        memory = {}
        warn("All memory has been reset!")
    end
end

-- Function to extract just the response text without prefix for memory
local function extractResponseText(fullResponse)
    -- Find the position of the separator in the response
    local separatorPos = string.find(fullResponse, _G.NAME_SEPARATOR)
    if separatorPos then
        -- Return everything after the separator and space
        return string.sub(fullResponse, separatorPos + 2) -- +2 to skip separator and space
    end
    -- If no separator found, return the original response (might be a memory recall response)
    return fullResponse
end

-- SMOOTH DRAGGING FUNCTIONALITY --
local function makeDraggable(frame)
    local dragging = false
    local dragStart, startPos

    UserInputService.InputBegan:Connect(function(input, gameProcessed)
        if gameProcessed then return end
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            local position = input.Position
            local guiPosition = frame.Position
            local guiSize = frame.Size
            local guiX = guiPosition.X.Scale * game:GetService("GuiService"):GetScreenResolution().X + guiPosition.X.Offset
            local guiY = guiPosition.Y.Scale * game:GetService("GuiService"):GetScreenResolution().Y + guiPosition.Y.Offset
            local guiWidth = guiSize.X.Scale * game:GetService("GuiService"):GetScreenResolution().X + guiSize.X.Offset
            local guiHeight = guiSize.Y.Scale * game:GetService("GuiService"):GetScreenResolution().Y + guiSize.Y.Offset

            if position.X >= guiX and position.X <= guiX + guiWidth and position.Y >= guiY and position.Y <= guiY + guiHeight then
                dragging = true
                dragStart = position
                startPos = frame.Position
            end
        end
    end)

    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = false
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
            local delta = input.Position - dragStart
            local newX = startPos.X.Offset + delta.X
            local newY = startPos.Y.Offset + delta.Y
            frame.Position = UDim2.new(startPos.X.Scale, newX, startPos.Y.Scale, newY)
        end
    end)
end

local chat = function(_string)
    warn("Attempting to send message: " .. _string)
    if game.TextChatService.ChatVersion == Enum.ChatVersion.TextChatService then
        print("FIRING ".._string.."...")
        local success, err = pcall(function()
            game.TextChatService.TextChannels.RBXGeneral:SendAsync(_string, "All")
        end)
        if not success then
            warn("Failed to send message via TextChatService: "..err)
        else
            warn("FIRED via TextChatService!")
        end
    else
        local success, err = pcall(function()
            game:GetService('ReplicatedStorage').DefaultChatSystemChatEvents.SayMessageRequest:FireServer(_string, 'All')
        end)
        if not success then
            warn("Failed to send message via legacy chat: "..err)
        else
            warn("FIRED via legacy chat!")
        end
    end
end

local RequestFunction = syn and syn.request or request

-- Ollama Request Function
local function MakeOllamaRequest(Prompt)
    return RequestFunction({
        Url = "http://localhost:11434/api/generate",
        Method = "POST",
        Headers = {
            ["Content-Type"] = "application/json"
        },
        Body = HttpService:JSONEncode({
            model = "gemma3:1b-it-qat", 
            prompt = Prompt,
            stream = false
        })
    })
end

-- Gemini Request Function
local function MakeGeminiRequest(Prompt)
    return RequestFunction({
        Url = "https://generativelanguage.googleapis.com/v1beta/models/" .. GEMINI_MODEL .. ":generateContent?key=" .. GEMINI_API_KEY,
        Method = "POST",
        Headers = {
            ["Content-Type"] = "application/json"
        },
        Body = HttpService:JSONEncode({
            contents = {{
                parts = {{
                    text = Prompt
                }}
            }},
            generationConfig = {
                maxOutputTokens = 100
            }
        })
    })
end

-- Groq Request Function
local function MakeGroqRequest(Prompt)
    return RequestFunction({
        Url = "https://api.groq.com/openai/v1/chat/completions",
        Method = "POST",
        Headers = {
            ["Content-Type"] = "application/json",
            ["Authorization"] = "Bearer " .. GROQ_API_KEY
        },
        Body = HttpService:JSONEncode({
            messages = {{
                role = "user",
                content = Prompt
            }},
            model = GROQ_MODEL,
            max_tokens = 100,
            temperature = 0.7
        })
    })
end

-- Local Hugging Face Request Function
local function MakeHuggingFaceRequest(Prompt)
    -- For local inference server (like text-generation-webui or similar)
    return RequestFunction({
        Url = HF_LOCAL_URL .. "/v1/completions",
        Method = "POST",
        Headers = {
            ["Content-Type"] = "application/json"
        },
        Body = HttpService:JSONEncode({
            prompt = Prompt,
            max_tokens = 100,
            temperature = 0.7,
            do_sample = true
        })
    })
end

-- Function to build conversation context from player's memory
local function buildConversationContext(playerName, currentMessage)
    local playerMemory = getPlayerMemory(playerName)
    local context = ""
    
    if #playerMemory > 0 then
        context = "Previous conversation with " .. playerName .. ":\n"
        for i, entry in ipairs(playerMemory) do
            context = context .. playerName .. ": " .. entry.message .. "\n"
            -- Just use the clean response without "You:" prefix
            context = context .. entry.response .. "\n"
        end
        context = context .. "Current message from " .. playerName .. ": " .. currentMessage
    else
        context = "First message from " .. playerName .. ": " .. currentMessage
    end
    
    return context
end

-- Function to create the final response with proper prefix
local function createFinalResponse(playerName, aiResponse)
    -- Dynamically set the prefix based on _G.PLAYER_NAME_FORMAT
    local prefix = ""
    
    if _G.PLAYER_NAME_FORMAT == "P" then
        -- Just player name with separator: "PlayerName: response"
        prefix = playerName .. _G.NAME_SEPARATOR .. " "
    elseif _G.PLAYER_NAME_FORMAT == "" then
        -- No prefix at all: "response"
        prefix = ""
    elseif string.sub(_G.PLAYER_NAME_FORMAT, 1, 2) == "P " then
        -- If starts with "P " then "PlayerName custom: response"
        local customText = string.sub(_G.PLAYER_NAME_FORMAT, 3)
        prefix = playerName .. " " .. customText .. _G.NAME_SEPARATOR .. " "
    else
        -- Custom prefix only: "qwe: response"
        prefix = _G.PLAYER_NAME_FORMAT .. _G.NAME_SEPARATOR .. " "
    end
    
    return prefix .. aiResponse
end

-- Function to process messages (shared between both chat systems)
local function processMessage(Message, player, Character)
    warn("Processing message from " .. player.Name .. ": " .. Message)
    if not AI_ACTIVE then
        warn("AI is not active, skipping message.")
        return
    end
    if player == LocalPlayer then
        warn("Message from local player, skipping.")
        return
    end
    if string.match(Message, "#") then
        warn("Message contains '#', skipping.")
        return
    end
    if not Character or not Character:FindFirstChild("Head") then
        warn("Player character or head not found, skipping.")
        return
    end
    if not LocalPlayer.Character or not LocalPlayer.Character:FindFirstChild("Head") then
        warn("Local player character or head not found, skipping.")
        return
    end
    if Debounce then
        warn("Debounce active, skipping.")
        return
    end
    if #Message < _G.MESSAGE_SETTINGS["MINIMUM_CHARACTERS"] or #Message > _G.MESSAGE_SETTINGS["MAXIMUM_CHARACTERS"] then
        warn("Message length out of bounds (" .. #Message .. "), skipping.")
        return
    end
    if CLOSE_RANGE_ONLY then 
        if _G.BLACKLISTED[player.Name] then
            warn("Player " .. player.Name .. " is blacklisted, skipping.")
            return
        end
        if (Character.Head.Position - LocalPlayer.Character.Head.Position).Magnitude > _G.MESSAGE_SETTINGS["MAXIMUM_STUDS"] then
            warn("Player " .. player.Name .. " is too far (" .. (Character.Head.Position - LocalPlayer.Character.Head.Position).Magnitude .. " studs), skipping.")
            return
        end
    elseif not _G.WHITELISTED[player.Name] then
        warn("Player " .. player.Name .. " is not whitelisted, skipping.")
        return
    end

    Debounce = true
    statusLabel.Text = "Status: Processing..."

    local Response = ""
    local playerName = player.Name
    
    if string.lower(Message) == "what did i just say" then
        local playerMemory = getPlayerMemory(playerName)
        if #playerMemory > 0 then
            local cleanResponse = "You just said: " .. playerMemory[#playerMemory].message
            Response = createFinalResponse(playerName, cleanResponse)
        else
            local cleanResponse = "I don't remember you saying anything recently."
            Response = createFinalResponse(playerName, cleanResponse)
        end
    else
        -- Build conversation context from this specific player's memory
        local context = buildConversationContext(playerName, Message)
        local Prompt = "(note: dont use any line breaks and keep the messages shorter) " .. context
        
        local HttpRequest
        if AI_PROVIDER == 0 then
            warn("Using Ollama API for " .. playerName .. "...")
            HttpRequest = MakeOllamaRequest(Prompt)
        elseif AI_PROVIDER == 1 then
            warn("Using Gemini API for " .. playerName .. "...")
            HttpRequest = MakeGeminiRequest(Prompt)
        elseif AI_PROVIDER == 2 then
            warn("Using Groq API for " .. playerName .. "...")
            HttpRequest = MakeGroqRequest(Prompt)
        else
            warn("Using Hugging Face Local for " .. playerName .. "...")
            HttpRequest = MakeHuggingFaceRequest(Prompt)
        end
        
        print("["..playerName.."]: "..Message)
        
        if HttpRequest.StatusCode == 200 then
            local success, responseData = pcall(function()
                return HttpService:JSONDecode(HttpRequest.Body)
            end)
            
            if success then
                local actualResponse
                
                if AI_PROVIDER == 0 then
                    -- Parse Ollama response
                    actualResponse = responseData.response
                elseif AI_PROVIDER == 1 then
                    -- Parse Gemini response
                    actualResponse = responseData.candidates and responseData.candidates[1] and 
                                   responseData.candidates[1].content and 
                                   responseData.candidates[1].content.parts and 
                                   responseData.candidates[1].content.parts[1] and 
                                   responseData.candidates[1].content.parts[1].text
                elseif AI_PROVIDER == 2 then
                    -- Parse Groq response
                    actualResponse = responseData.choices and responseData.choices[1] and 
                                   responseData.choices[1].message and 
                                   responseData.choices[1].message.content
                else
                    -- Parse Hugging Face local response
                    actualResponse = responseData.choices and responseData.choices[1] and responseData.choices[1].text
                end

                if actualResponse == nil or actualResponse == "" then
                    warn("AI response is empty or nil")
                    Response = createFinalResponse(playerName, "Hmm, let me think about that...")
                else 
                    -- Clean up the response
                    actualResponse = string.gsub(actualResponse, "\n", " ")
                    actualResponse = string.gsub(actualResponse, "  ", " ")
                    Response = createFinalResponse(playerName, actualResponse)
                    warn("Sending: "..Response)
                end
            else
                warn("Failed to parse JSON response: " .. HttpRequest.Body)
                Response = createFinalResponse(playerName, "Sorry, I had trouble understanding that.")
            end
        else
            warn("Request failed with status code: " .. HttpRequest.StatusCode)
            if AI_PROVIDER == 2 then
                if HttpRequest.StatusCode == 401 then
                    Response = createFinalResponse(playerName, "Groq API key error. Check settings.")
                elseif HttpRequest.StatusCode == 429 then
                    Response = createFinalResponse(playerName, "Rate limit exceeded. Try again later.")
                else
                    Response = createFinalResponse(playerName, "Sorry, I couldn't respond right now.")
                end
            elseif AI_PROVIDER == 3 then
                Response = createFinalResponse(playerName, "Local AI not available. Check server.")
            else
                Response = createFinalResponse(playerName, "Sorry, I couldn't respond. Try again!")
            end
        end
    end

    -- Add to this specific player's memory (store just the AI response text, not the full prefixed message)
    local responseTextOnly = extractResponseText(Response)
    addToMemory(playerName, Message, responseTextOnly)

    chat(string.sub(Response, 1, 128))
    if #Response > 128 then
        delay(1, function()
            chat(string.sub(Response, 129))
        end)
    end
    wait()
    statusLabel.Text = AI_ACTIVE and "Status: Active" or "Status: Inactive"
    Debounce = false
end

-- Detect messages using TextChatService (new chat system)
if TextChatService.ChatVersion == Enum.ChatVersion.TextChatService then
    local channel = TextChatService.TextChannels.RBXGeneral
    if not channel then
        warn("RBXGeneral channel not found. Attempting to find another channel...")
        for _, chan in pairs(TextChatService.TextChannels:GetChildren()) do
            if chan:IsA("TextChannel") then
                channel = chan
                warn("Found channel: " .. chan.Name)
                break
            end
        end
    end
    if channel then
        warn("Listening for messages on channel: " .. channel.Name)
        channel.MessageReceived:Connect(function(message)
            warn("Message received via TextChatService: " .. message.Text)
            local sender = message.TextSource
            if not sender then
                warn("No sender found for message.")
                return
            end
            local player = Players:GetPlayerByUserId(sender.UserId)
            if not player then
                warn("No player found for sender ID: " .. sender.UserId)
                return
            end
            local Message = message.Text
            local Character = player.Character
            processMessage(Message, player, Character)
        end)
    else
        warn("No TextChatService channel found. New chat system will not work.")
    end
end

-- Detect messages using legacy chat system (2017 chat)
local success, err = pcall(function()
    local OnMessageDoneFiltering = ReplicatedStorage:WaitForChild("DefaultChatSystemChatEvents", 5):WaitForChild("OnMessageDoneFiltering", 5)
    warn("Legacy chat system found, setting up listener...")
    OnMessageDoneFiltering.OnClientEvent:Connect(function(Table)
        warn("Message received via legacy chat: " .. Table.Message)
        local Message, Instance = Table.Message, Players:FindFirstChild(Table.FromSpeaker)
        local Character = Instance and Instance.Character
        processMessage(Message, Instance, Character)
    end)
end)
if not success then
    warn("Failed to set up legacy chat listener: " .. err)
end
